'use client'

import { useState, useEffect } from 'react'
import { X, Mail, Send } from 'lucide-react'

export function NewsletterModal({ onClose }: { onClose: () => void }) {
  const [email, setEmail] = useState('')

  useEffect(() => { document.body.style.overflow = 'hidden'; return () => { document.body.style.overflow = 'unset' } }, [])

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    if (email) { localStorage.setItem('subscribed', 'true'); alert('Thanks for subscribing!'); onClose() }
  }

  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center p-4 bg-black/80 backdrop-blur-sm">
      <div className="relative max-w-md w-full bg-gradient-to-br from-moon-card to-moon-dark rounded-2xl moon-border p-6">
        <button onClick={onClose} className="absolute top-4 right-4 p-1 rounded-full hover:bg-white/10"><X className="w-5 h-5" /></button>
        <div className="text-center mb-6">
          <div className="text-5xl mb-3 animate-float">🌙</div>
          <h3 className="text-2xl font-bold mb-2">Join THE MOON</h3>
          <p className="text-moon-gray text-sm">Get the best stories delivered to your inbox.</p>
        </div>
        <form onSubmit={handleSubmit} className="space-y-4">
          <div className="relative">
            <Mail className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-moon-gray" />
            <input type="email" placeholder="Your email" value={email} onChange={(e) => setEmail(e.target.value)}
              className="w-full pl-10 pr-4 py-3 rounded-xl bg-moon-dark moon-border text-white focus:outline-none focus:border-moon-blue" required />
          </div>
          <button type="submit" className="w-full py-3 rounded-xl bg-moon-blue text-black font-semibold hover:bg-blue-400 flex items-center justify-center gap-2">
            Subscribe <Send className="w-4 h-4" />
          </button>
        </form>
      </div>
    </div>
  )
}

'use client'

import { useState, useEffect } from 'react'
import { Zap } from 'lucide-react'

const breakingNews = [
  "BREAKING: Global AI summit reaches historic agreement",
  "JUST IN: Major quantum computing breakthrough achieved",
  "ALERT: Historic climate deal signed by 150 nations"
]

export function BreakingTicker() {
  const [currentIndex, setCurrentIndex] = useState(0)

  useEffect(() => {
    const interval = setInterval(() => {
      setCurrentIndex((prev) => (prev + 1) % breakingNews.length)
    }, 5000)
    return () => clearInterval(interval)
  }, [])

  return (
    <div className="fixed top-16 left-0 right-0 z-40 bg-gradient-to-r from-red-600/90 to-moon-blue/90 backdrop-blur-lg">
      <div className="container-custom py-2.5">
        <div className="flex items-center gap-3">
          <div className="flex items-center gap-2 bg-white/20 px-3 py-1 rounded-full">
            <Zap className="w-4 h-4 text-yellow-400 animate-pulse" />
            <span className="text-xs font-bold">Breaking</span>
          </div>
          <p className="text-sm font-medium truncate">{breakingNews[currentIndex]}</p>
        </div>
      </div>
    </div>
  )
}

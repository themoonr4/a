'use client'

import { useState, useEffect } from 'react'
import Image from 'next/image'
import { ChevronLeft, ChevronRight } from 'lucide-react'

const featuredStories = [
  { id: 1, title: "The Dawn of Artificial Intelligence", excerpt: "How AI is reshaping our world.",
    image: "https://images.unsplash.com/photo-1677442136019-21780ecad995", category: "Technology" },
  { id: 2, title: "Mars Colony: The New Frontier", excerpt: "Humanity's first permanent settlement on Mars.",
    image: "https://images.unsplash.com/photo-1614313913007-2d4e3db9bf1a", category: "Science" },
]

export function HeroCarousel() {
  const [current, setCurrent] = useState(0)

  useEffect(() => {
    const timer = setInterval(() => setCurrent((prev) => (prev + 1) % featuredStories.length), 6000)
    return () => clearInterval(timer)
  }, [])

  return (
    <div className="relative h-[500px] rounded-2xl overflow-hidden group">
      <Image src={featuredStories[current].image} alt={featuredStories[current].title} fill className="object-cover" />
      <div className="absolute inset-0 bg-gradient-to-t from-moon-dark via-transparent to-transparent" />
      
      <div className="absolute bottom-0 left-0 right-0 p-8">
        <span className="inline-block px-3 py-1 rounded-full bg-moon-blue/20 text-moon-blue text-xs mb-4">
          {featuredStories[current].category}
        </span>
        <h2 className="text-4xl font-bold mb-2">{featuredStories[current].title}</h2>
        <p className="text-moon-gray">{featuredStories[current].excerpt}</p>
      </div>

      <button onClick={() => setCurrent((prev) => (prev - 1 + featuredStories.length) % featuredStories.length)}
        className="absolute left-4 top-1/2 p-2 rounded-full bg-black/50 opacity-0 group-hover:opacity-100">
        <ChevronLeft className="w-6 h-6" />
      </button>
      <button onClick={() => setCurrent((prev) => (prev + 1) % featuredStories.length)}
        className="absolute right-4 top-1/2 p-2 rounded-full bg-black/50 opacity-0 group-hover:opacity-100">
        <ChevronRight className="w-6 h-6" />
      </button>
    </div>
  )
}
